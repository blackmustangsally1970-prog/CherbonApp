    function prefixMatcher(params, data) {
        if (!params.term) return data;
        if (!data.text) return null;

        const term = params.term.toLowerCase();
        const text = data.text.toLowerCase();

        return text.startsWith(term) ? $.extend({}, data, true) : null;
    }

function isValidDate(dateStr) {
    const d = new Date(dateStr + "T00:00:00");
    return !isNaN(d.getTime());
}

function hasDuplicateDates(lessons) {
    const dates = lessons.map(l => l.lesson_date);
    return new Set(dates).size !== dates.length;
}

function canonicalTimeFrame(tf) {
    if (!tf) return "";

    // Remove spaces
    tf = tf.replace(/\s+/g, "");

    // Split
    const [start, end] = tf.split("-");

    // Zero-pad
    const pad = t => {
        let [h, m] = t.split(":");
        h = h.padStart(2, "0");
        m = m.padStart(2, "0");
        return `${h}:${m}`;
    };

    return `${pad(start)} - ${pad(end)}`;
}


// ⭐ Unified Save Handler (Option A — send time_frame only)
async function actuallySaveLessons() {
    console.log("HANDLER ENTERED");

    const info = collectBookingInfo();
    const lessons = collectLessonRows();

    if (lessons.length === 0) {
        alert("No lessons to save.");
        return;
    }

// ⭐ Validate rows
for (const row of lessons) {

    // ⭐ Allow Payment + Voucher CR to skip time
    if (row.lesson_type !== "Payment" && row.lesson_type !== "Voucher CR") {
        if (!row.lesson_date || !row.time_frame) {
            alert("Each lesson requires a date and time.");
            return;
        }
    }

    if (!isValidDate(row.lesson_date)) {
        alert("One or more dates are invalid.");
        return;
    }
}

    const freq = document.getElementById("recurrence_type").value;

    // ============================================================
    // ⭐ EXISTING CLIENT MODE
    // ============================================================
    if (info.booking_type === "existing") {

        if (!info.client_id) {
            alert("Please select an existing client.");
            return;
        }

        for (const row of lessons) {

            const form = new FormData();
            form.append("date", row.lesson_date);

            // ⭐ Send full time frame (backend splits)
            form.append("time_frame", canonicalTimeFrame(row.time_frame));



            form.append("lesson_type", row.lesson_type);
            form.append("group_priv", row.group_priv);
            form.append("price_pl", row.price);
            form.append("payment", row.payment);
            form.append("recurrence_type", freq);
            form.append("client_mode", "existing");
            form.append("client_id", info.client_id);

            form.append("respect_blockouts", document.getElementById("respect_blockouts").value);



            await fetch("/new_lesson", {
                method: "POST",
                body: form
            });
        }

        alert("Existing client booking saved!");
        return;
    }

    // ============================================================
    // ⭐ NEW CLIENT MODE
    // ============================================================
    if (info.booking_type === "new") {

        if (!info.new_client_name) {
            alert("Please enter the new client's name.");
            return;
        }

        for (const row of lessons) {

            const form = new FormData();
            form.append("date", row.lesson_date);

            // ⭐ Send full time frame
            form.append("time_frame", canonicalTimeFrame(row.time_frame));

            form.append("lesson_type", row.lesson_type);
            form.append("group_priv", row.group_priv);
            form.append("price_pl", row.price);
            form.append("payment", row.payment);
            form.append("recurrence_type", freq);

            form.append("client_mode", "new");
            form.append("client_name", info.new_client_name);
            form.append("client_phone", info.mobile);

            form.append("respect_blockouts", document.getElementById("respect_blockouts").value);



            await fetch("/new_lesson", {
                method: "POST",
                body: form
            });
        }

        alert("New client booking saved!");
        return;
    }

    // ============================================================
    // ⭐ SMS MODE
    // ============================================================
    if (info.booking_type === "sms") {

        if (!info.mobile) {
            alert("Mobile number is required for SMS invites.");
            return;
        }

        const btn = document.getElementById("save-lessons-btn");
        if (btn) {
            btn.disabled = true;
            btn.textContent = "Sending SMS Invites...";
        }

        for (const row of lessons) {

            const payload = new FormData();
            payload.append("lesson_date", row.lesson_date);
            payload.append("time_frame", row.time_frame);
            payload.append("lesson_type", row.lesson_type);
            payload.append("group_priv", row.group_priv);
            payload.append("mobile", info.mobile);
            payload.append("riders", row.riders);
            payload.append("price", row.price);

            await fetch("/send_invite", {
                method: "POST",
                body: payload
            });
        }

        if (btn) {
            btn.textContent = "Invites Sent!";
            setTimeout(() => {
                btn.disabled = false;
                btn.textContent = "Save Lessons";
            }, 1500);
        }

        return;
    }

}

// ⭐ Blockout-aware wrapper for Save Lessons
document.getElementById("save-lessons-btn").addEventListener("click", async (e) => {
    e.preventDefault();

    const respect = document.getElementById("respect_blockouts").value;
    const startDate = document.getElementById("top_base_date").value;
    const freq = document.getElementById("recurrence_type").value;

    // ⭐ If override → skip popup entirely
    if (respect === "no") {
        return actuallySaveLessons();
    }

    // ⭐ No date? Just save
    if (!startDate) {
        return actuallySaveLessons();
    }

    // ⭐ Call backend blockout checker
    const resp = await fetch("/api/check_blockouts", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
            start_date: startDate,
            freq: freq
        })
    });

    const data = await resp.json();

    // ⭐ No blockouts → save normally
    if (!data.blocked) {
        return actuallySaveLessons();
    }

    // ⭐ Blockouts found → show popup
    showBlockoutPopup(data.blocked_dates);
});

function showBlockoutPopup(blockedDates) {
    const modal = document.getElementById("blockout-warning-modal");
    const list = document.getElementById("blockout-dates-list");

    list.innerHTML = "";

    blockedDates.forEach(item => {
        const li = document.createElement("li");

        // ⭐ Colour logic
        if (item.reason.toLowerCase().includes("single")) {
            // 🔴 Single‑day blockout
            li.style.background = "#ffdddd";          // light red
            li.style.borderLeft = "4px solid #d00";   // strong red
        } else {
            // 🟡 Range blockout
            li.style.background = "#fff7cc";          // light yellow
            li.style.borderLeft = "4px solid #e6b800"; // gold
        }

        li.style.padding = "6px 10px";
        li.style.marginBottom = "6px";
        li.style.borderRadius = "4px";
        li.style.listStyle = "none";

        li.textContent = `${item.date} — ${item.reason}`;
        list.appendChild(li);
    });

    modal.style.display = "flex";
}
